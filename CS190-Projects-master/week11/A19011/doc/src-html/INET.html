063    public String getFromFile(String strFileName) throws Exception {

104    /**
105     * Will make a pattern and will try to find a matching part in the StrInput
106     * @param strInput The text to search in
107     * @param strPattern The pattern to try to find
108     * @return Returns the match Regular Exprestion or null if it does not find anything
109     */
110    public String getRegEx(String strInput, String strPattern) {
111        //Makes a new pattern
112        Pattern p = Pattern.compile(strPattern);
113        //Finds a match to the pattern
114        Matcher matcher = p.matcher(strInput);
115        //If it matches it will return the group
116        if(matcher.matches()){
117            return matcher.group();
118        }
119        else{
120            return null;
121        }
122    }
