015public class calc2 extends JDialog {
016    private JPanel contentPane;
017    private JButton buttonOK;
018    private JButton buttonCancel;
019    private JTextField AircraftEmptyWeight;
020    private JTextField BaggageWeight;
021    private JTextField GallonsFuel;
022    private JTextField NumPassFront;
023    private JTextField NumPassBack;
024    private JLabel RampWeightAircraft;
025    final double WT_PERSON = 190.0;
026    final double WT_AVGAS = 6.02;
027
028    public calc2() {
029        setContentPane(contentPane);
030        setModal(true);
031        getRootPane().setDefaultButton(buttonOK);
032
033        buttonOK.addActionListener(new ActionListener() {
034            public void actionPerformed(ActionEvent e) {
035                onOK();
036            }
037        });
038
039        buttonCancel.addActionListener(new ActionListener() {
040            public void actionPerformed(ActionEvent e) {
041                onCancel();
042            }
043        });
044
045
046// call onCancel() when cross is clicked
047        setDefaultCloseOperation(DO_NOTHING_ON_CLOSE);
048        addWindowListener(new WindowAdapter() {
049            public void windowClosing(WindowEvent e) {
050                onCancel();
051            }
052        });
053
054// call onCancel() on ESCAPE
055        contentPane.registerKeyboardAction(new ActionListener() {
056            public void actionPerformed(ActionEvent e) {
057                onCancel();
058            }
059        }, KeyStroke.getKeyStroke(KeyEvent.VK_ESCAPE, 0), JComponent.WHEN_ANCESTOR_OF_FOCUSED_COMPONENT);
060
061
062    }
063
064    private void calcForm(){
065
066        double temp,aircraftempty, weightPass, FuelWeight, baggageWeight;
067        aircraftempty = Double.parseDouble(AircraftEmptyWeight.getText());
068        weightPass = (Double.parseDouble(NumPassBack.getText()) + Double.parseDouble(NumPassFront.getText())+1)*WT_PERSON;
069        FuelWeight = (Double.parseDouble(GallonsFuel.getText())* WT_AVGAS);
070        baggageWeight= Double.parseDouble(BaggageWeight.getText());
071        RampWeightAircraft.setText(String.valueOf(aircraftempty+weightPass+FuelWeight+baggageWeight)+"pounds");
072
073
074    }
075
076
077
078    private void onOK() {
079// add your code here
080        calcForm();
081    }
082
083    private void onCancel() {
084// add your code here if necessary
085        dispose();
086    }
087
088    public static void main(String[] args) {
089        calc2 dialog = new calc2();
090        dialog.pack();
091        dialog.setVisible(true);
092        System.exit(0);
093    }
094}
