021    /**
022     * Will set up a scanner to read the file and add each station line into the arraylist staList as well as add just the station ID to arraylist staIndex
023     *
024     * @param strFbinPath The input file pathway for the scanner to read the FBIN
025     */
026    public WorldStations(String strFbinPath, String strWorldPath) {
027        //calls the super
028        super(strFbinPath);
029        //makes a new stations object
030        st = new Stations("data\\FBIN.txt");
031        //test
032        //System.out.println(sta.NAME);
033
034        //makes a new String array for aryWorld
035        aryWorld = new String[st.length()];
036        //makes a new file called f
037        File f = new File("data\\World.txt");
038        try {
039            //makes a scanner
040            Scanner scan = new Scanner(f);
041            //makes a temp string
042            String tempz = "";
043            //while scanner has a next line
044            while (scan.hasNext()) {
045                //if the scan has a next line, extra line of defense
046                if (scan.hasNext()) {
047                    //temp now = the next line of scan
048                    tempz = scan.nextLine();
049                    //If the first 2 numbers are 73 or 74 which is the code for the USA
050                    if (tempz.substring(0, 2).equals("72") || tempz.substring(0, 2).equals("74")) {
051                        //if the staIndex arraylist in the stations object contains the same station ID of the current line of the world.txt file.
052                        if (st.staIndex.contains(tempz.substring(8, 11))) {
053                            //adds the whole tempz string into the corrisponding index in the aryWorld array
054                            aryWorld[st.staIndex.indexOf(tempz.substring(8, 11))] = tempz;
055                        }
056                    }
057                }
058            }
059        } catch (FileNotFoundException e) {
060            e.printStackTrace();
061        }
062
063
064    }
