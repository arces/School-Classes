094    /**
095     * Converts the Alt to a usable format for NWSFB05
096     *
097     * @param str The alt from the combo box
098     * @return The Alt that can be used by NWSFB05
099     */
100    public String convertAlt(String str) {
101        if (str.equals("3000")) {
102            return "03";
103        } else if (str.equals("6000")) {
104            return "06";
105        } else if (str.equals("9000")) {
106            return "09";
107        } else if (str.equals("12000")) {
108            return "12";
109        } else if (str.equals("18000")) {
110            return "18";
111        } else if (str.equals("24000")) {
112            return "24";
113        } else if (str.equals("30000")) {
114            return "30";
115        } else if (str.equals("34000")) {
116            return "34";
117        } else if (str.equals("39000")) {
118            return "39";
119        } else {
120            return "";
121        }
122    }

124    /**
125     * Sets the Lables based on info from the combo boxes and FBIN.txt
126     */
127    public void setLables() {
128        //Gets the Station ID from the Combo Box
129        String sta = String.valueOf(cbLocation.getSelectedItem());
130        //Makes a new NWSFB05 object
131        NWSFB05 nw = new NWSFB05(st.getStaWea(sta));
132
133        //gets the alt from the combo box
134        String alt = convertAlt(String.valueOf(cbAlt.getSelectedItem()));
135
136        //Makes temp vars to be used
137        String WindTemp = "";
138        String Celsius = "C";
139
140        //Sees if the Temp is N/A and if it is then it does not add the "C" at the end
141        if ("N/A".equals(nw.getWindTemp(alt))) {
142            WindTemp = nw.getWindTemp(alt);
143        } else {
144            WindTemp = nw.getWindTemp(alt) + Celsius;
145        }
146
147        //sets the labels to the correct info
148        lbWindDir.setText(nw.getWindDir(alt));
149        lbWindSpeed.setText(nw.getWindSpeed(alt));
150        lbWindTemp.setText(WindTemp);
151    }
