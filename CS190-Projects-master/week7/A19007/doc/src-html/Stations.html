001import java.io.File;
002import java.io.FileNotFoundException;
003import java.io.PrintWriter;
004import java.util.ArrayList;
005import java.util.Scanner;
006
007/**
008 * Created by star on 10/3/2015.
009 */
010public class Stations {
011
012    public ArrayList<String> staList = new ArrayList<String>();
013    public ArrayList<String> staIndex = new ArrayList<String>();
014    public PrintWriter pw;
015    public boolean breakLoop = false;
016    public String SID;
017
018
019    /**
020     * Will set up a scanner to read the file and add each station line into the arraylist staList as well as add just the station ID to arraylist staIndex
021     * @param temp The input file pathway for the scanner to read the FBIN
022     */
023    public Stations(String temp) {
024        String strFilePath = temp;
025        File textfile = new File(strFilePath);
026        File outFile = new File("out\\FBOUT.txt");
027        try {
028            //makes a new printWriter
029            pw = new PrintWriter(outFile);
030            //makes a new var to count the loops
031            int x = 0;
032            try {
033                Scanner scan = new Scanner(textfile);
034
035                //will loop until there is no more lines to scan
036                while (breakLoop == false) {
037                    //adds the next scan line into String tempz
038                    String tempz = scan.nextLine();
039                    //checks to see if there is a next line and that it looped at least 7 times
040                    if (scan.hasNext() && x > 6) {
041                        x++;
042                        //add the whole line into staList
043                        staList.add(tempz);
044                        //adds just the station ID to staIndex
045                        staIndex.add(tempz.substring(0, 3));
046
047                    }
048                    //checks to see it is in the first 6 lines of the file and will skip them
049                    else if (scan.hasNext() && x < 7) {
050                        x++;
051                    }
052                    //ends the loop, there are no more lines in the file
053                    else {
054                        breakLoop = true;
055                    }
056                }
057
058                //closes the file
059                scan.close();
060
061            } catch (FileNotFoundException e) {
062                e.printStackTrace();
063            }
064
065
066        } catch (FileNotFoundException e) {
067            e.printStackTrace();
068        }
069
070    }
071
072    /**
073     * Checks to see if the Station ID is in the ArrayList and will return true or false
074     * @param strID The station ID as a string
075     * @return returns true or false depending on if the ID apperes or not
076     */
077    public boolean exists(String strID) {
078        return staIndex.contains(strID);
079    }
080
081    /**
082     * Gets the length of the ArrayList staIndex
083     * @return Returns the length of the StaIndex in an Int
084     */
085    public int length() {
086        return staIndex.size();
087    }
088
089    /**
090     * Takes in a string of the Station ID and returns the full string
091     * @param strSationID The station ID as a String
092     * @return returns the whole Station data for later processing
093     */
094
095    public String getStaWea(String strSationID) {
096
097        if (exists(strSationID)) {
098            //gets the index from staindex and gets the string in stalist of the corresponding index
099            return staList.get(staIndex.indexOf(strSationID));
100        } else {
101            //If the station ID is not valid then it returns Station Not Found
102            return "Station Not Found";
103        }
104    }
105
106    /**
107     * Will write what ever it is passed to the Print Write file
108     * @param s Any input that will be written to the PW file as a String
109     */
110    public void writeToFile(String s) {
111        //adds S to the pw file on a new line
112        pw.println(s);
113    }
114
115    public String getStationID(int intPosition){
116        return staIndex.get(intPosition);
117    }
118
119    /**
120     * Will close the Print Write file
121     */
122    public void closePW() {
123        //closes the pw file
124        pw.close();
125    }
